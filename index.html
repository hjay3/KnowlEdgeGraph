// Orb tilt + auto-orbit
const orb = document.getElementById('orb');
let tiltX=0, tiltY=0, tx=0, ty=0;
let auto=true, angle=0, speed=18;
function update(){
  if(auto){ angle += speed*0.002; tx = Math.sin(angle)*26; ty = Math.cos(angle*1.2)*18; }
  tiltX += (tx-tiltX)*0.08; tiltY += (ty-tiltY)*0.08;
  orb.style.transform = `rotateX(${tiltY.toFixed(2)}deg) rotateY(${tiltX.toFixed(2)}deg)`;
  requestAnimationFrame(update);
}
update();
